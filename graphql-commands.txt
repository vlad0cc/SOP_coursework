GraphiQL URL:
http://localhost:8080/graphiql

Result locations:
- GraphQL response: right panel in GraphiQL
- Audit events: http://localhost:8081/api/audit?limit=50
- Notifications UI: http://localhost:8084/

1. Проверка стартовых данных

query InitialData {
  employees(page: 0, size: 10) {
    totalElements
    content {
      id
      fullName
      position
      documentsCount
    }
  }
  documents(page: 0, size: 10) {
    totalElements
    content {
      id
      title
      documentNumber
      description
      status
      auditComment
      signature
      sender { id fullName }
      recipient { id fullName }
    }
  }
}

4. Проверка фильтров документов

query DocumentFilters {
  bySender: documents(filter: { senderId: "1" }, page: 0, size: 10) {
    totalElements
    content {
      id
      title
      documentNumber
      sender { id fullName }
    }
  }

  byRecipient: documents(filter: { recipientId: "2" }, page: 0, size: 10) {
    totalElements
    content {
      id
      title
      documentNumber
      recipient { id fullName }
    }
  }

  byStatus: documents(filter: { status: "SENT" }, page: 0, size: 10) {
    totalElements
    content {
      id
      title
      status
    }
  }

  byTitle: documents(filter: { titleSearch: "заявка" }, page: 0, size: 10) {
    totalElements
    content {
      id
      title
      documentNumber
    }
  }
}
